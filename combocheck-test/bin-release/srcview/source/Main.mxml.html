<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml"
    creationComplete="onCreationComplete()" xmlns:arcadiocarballares="com.arcadiocarballares.*" viewSourceURL="srcview/index.html">

    <mx:Script>
        <![CDATA[
            import mx.controls.CheckBox;
            import mx.controls.ComboBox;
            import mx.controls.Alert;
            import mx.collections.ArrayCollection;
            import mx.events.DropdownEvent;
            
            [Bindable]
            private var items:ArrayCollection=new ArrayCollection([
                {label:'label 1', value:1, assigned:true},
                {label:'label 2', value:2},
                {label:'label 3', value:3, assigned:true},
                {label:'label 4', value:4},
                {label:'label 5', value:5}
            ]);
            [Bindable]
            private var myText:String='';

            private function onCreationComplete():void {
                myCombo.selectedItem=null;
                onAddItem();    
            }
            
            private function onAddItem ():void {
                myText='';
                var items:ArrayCollection=myCombo.selectedItems
                for (var i:String in items) {
                    myText+=items[i].label+' ';
                }
            }
        ]]>
    </mx:Script>
    <mx:VBox width="100%" height="100%" horizontalAlign="center" verticalAlign="top">
        <mx:HBox>
            <mx:Label text="SelectedItems:"/>
            <mx:Text text="{myText}"/>
        </mx:HBox>
        <mx:HBox>
            <mx:Label text="SelectedItem:"/>
            <mx:Text text="{myCombo.selectedItem.label}"/>
        </mx:HBox>
        <arcadiocarballares:ComboCheck id="myCombo" dataProvider="{items}" labelField="label" width="100" 
            addItem="onAddItem()"/>
    </mx:VBox>
</mx:Application>
