package com.acm
{
	import spark.components.supportClasses.DropDownController;
	
	[Event(name="check", type="com.acm.ComboCheckEvent")]
	
	public class ComboCheckDropDownController extends DropDownController
	{
		public function ComboCheckDropDownController()
		{
			super();
			addEventListener (ComboCheckEvent.CHECK, onCheck);
		}
		
		override public function closeDropDown(commit:Boolean):void {
			super.closeDropDown(commit);
			trace ("close");
		}
		
		public function onCheck (event:ComboCheckEvent):void {
			var evt:ComboCheckEvent = new ComboCheckEvent(ComboCheckEvent.CHECK, true);
			evt.obj=event.obj;
			dispatchEvent(evt);
		}
	}
}
